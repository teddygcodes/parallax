const BASE = '/api'

export async function fetchEvents(limit = 100, offset = 0): Promise<EventsResponse> {
  const res = await fetch(`${BASE}/events?limit=${limit}&offset=${offset}`, {
    cache: 'no-store',

### frontend/components/Globe.tsx

CRITICAL RULES:
1. This is a normal `'use client'` component. Do NOT put a `dynamic()` wrapper inside this file.
2. Globe.tsx must only run on the client. The `ssr: false` wrapper in page.tsx handles SSR prevention. Do not add a second `dynamic()` here.
3. globe.gl may be imported dynamically inside `useEffect` or used in any client-safe way — it must never execute during SSR.

```tsx
'use client'

import { useEffect, useRef, useState } from 'react'
import type { ConflictEvent } from '@/types'
import { fetchEvents } from '@/lib/api'

type GlobeState = 'loading' | 'error' | 'ready' | 'empty'

// Map event type to display color
function eventColor(type: ConflictEvent['event_type']): string {
  switch (type) {
    case 'STRIKE':  return '#c0392b'
    case 'MISSILE': return '#c0392b'
    case 'DRONE':   return '#8e44ad'
    case 'NAVAL':   return '#d4a017'
    case 'TROOP':   return 'rgba(232,230,224,0.15)'
    default:        return '#8a8a8a'
  }
}

export interface GlobeProps {
  onEventClick?: (event: ConflictEvent) => void
  onEventHover?: (event: ConflictEvent | null, x: number, y: number) => void
  timeWindowHours?: number   // undefined = no time filter
  visibleTypes?: Set<ConflictEvent['event_type']>  // undefined = all types shown; empty Set = nothing shown
}

export default function GlobeComponent({
  onEventClick,
  onEventHover,
  timeWindowHours,
  visibleTypes,
}: GlobeProps) {
  const containerRef = useRef<HTMLDivElement>(null)
  const globeRef = useRef<any>(null)
  const intervalRef = useRef<ReturnType<typeof setInterval> | null>(null)
  const [state, setState] = useState<GlobeState>('loading')
  const [allEvents, setAllEvents] = useState<ConflictEvent[]>([])

  // Filter events by current props — state MUST be derived from filtered result
  function applyFilters(events: ConflictEvent[]): ConflictEvent[] {
    let filtered = events

    if (timeWindowHours !== undefined) {
      const cutoff = Date.now() - timeWindowHours * 3_600_000
      filtered = filtered.filter(e => new Date(e.first_detection_time).getTime() >= cutoff)
    }

    if (visibleTypes !== undefined) {
      // Empty Set means all toggles off → render nothing
      if (visibleTypes.size === 0) return []
      filtered = filtered.filter(e => visibleTypes.has(e.event_type))
    }

    return filtered
  }

  async function loadAndRender(): Promise<ConflictEvent[]> {
    try {
      const data = await fetchEvents(100)
      setAllEvents(data.events)
      return data.events
    } catch {
      setState('error')
      return []
    }
  }

  function renderToGlobe(globe: any, events: ConflictEvent[]) {
    const filtered = applyFilters(events)

    // Drive display state from filtered set, not raw fetch count
    if (filtered.length === 0) {
      setState('empty')
    } else {
      setState('ready')
    }

    const points = filtered.filter(e => e.event_type !== 'MISSILE')
    const arcs   = filtered.filter(e => e.event_type === 'MISSILE')

    globe
      .pointsData(points)
      .pointLat((e: ConflictEvent) => e.lat)
      .pointLng((e: ConflictEvent) => e.lng)
      .pointColor((e: ConflictEvent) => eventColor(e.event_type))
      .pointAltitude((e: ConflictEvent) => e.event_type === 'DRONE' ? 0.005 : 0.01)
      .pointRadius((e: ConflictEvent) => e.event_type === 'DRONE' ? 0.4 : 0.5)
      .pointLabel(() => '')  // no default tooltip — HoverCard handles display

    // MISSILE arcs
    // v1 VISUAL PLACEHOLDER: MISSILE events lack real launch-point data.
    // Using synthetic short offset origin to produce visible arcs.
    // Real trajectory data (from ACLED/source) is a v2 improvement.
    globe
      .arcsData(arcs)
      .arcStartLat((e: ConflictEvent) => e.lat - 1.5)
      .arcStartLng((e: ConflictEvent) => e.lng - 1.5)
      .arcEndLat((e: ConflictEvent) => e.lat)
      .arcEndLng((e: ConflictEvent) => e.lng)
      .arcColor(() => '#c0392b')
      .arcDashLength(0.5)
      .arcDashGap(0.2)
      .arcDashAnimateTime(4000)
      .arcStroke(0.5)
  }

  useEffect(() => {
    if (!containerRef.current) return

    let mounted = true

    async function init() {
      setState('loading')

      const evts = await loadAndRender()
      if (!mounted) return

      // Dynamically import globe.gl — must never run on server
      const GlobeLib = (await import('globe.gl')).default
      if (!mounted || !containerRef.current) return

      // Globe surface: use a dark earth texture to achieve the desired
      // navy ocean / subdued landmass look. Do NOT mutate internal mesh
      // material properties — that's fragile and implementation-dependent.
      const globe = GlobeLib()(containerRef.current)
        .width(window.innerWidth)
        .height(window.innerHeight)
        .backgroundColor('#0a0a0e')
        .globeImageUrl('//unpkg.com/three-globe/example/img/earth-dark.jpg')
        .atmosphereColor('#1a3a5c')
        .atmosphereAltitude(0.15)

      globeRef.current = globe

      // Time-based rotation — barely perceptible.
      // 0.3 is the starting value. If rotation feels too active in-browser,
      // reduce toward 0.1. Never use requestAnimationFrame + manual increments.
      globe.controls().autoRotate = true
      globe.controls().autoRotateSpeed = 0.3

      // Hover and click handlers (wired up fully in Loop 5)
      globe.onPointHover((point: any, _prev: any, coords: { x: number; y: number }) => {
        onEventHover?.(point as ConflictEvent | null, coords.x, coords.y)
      })
      globe.onPointClick((point: any) => {
        if (point) onEventClick?.(point as ConflictEvent)
      })

      renderToGlobe(globe, evts)
    }

    init()

    // 30s auto-refresh
    intervalRef.current = setInterval(async () => {
      const evts = await loadAndRender()
      if (globeRef.current) renderToGlobe(globeRef.current, evts)
    }, 30_000)

    const handleResize = () => {
      if (globeRef.current) {
        globeRef.current.width(window.innerWidth).height(window.innerHeight)
      }
    }
    window.addEventListener('resize', handleResize)

    return () => {
      mounted = false
      if (intervalRef.current) clearInterval(intervalRef.current)
      window.removeEventListener('resize', handleResize)
    }
  }, [])  // eslint-disable-line react-hooks/exhaustive-deps

  // Re-render when filters change (without re-fetching)
  useEffect(() => {
    if (globeRef.current) renderToGlobe(globeRef.current, allEvents)
  }, [timeWindowHours, visibleTypes])  // eslint-disable-line react-hooks/exhaustive-deps

  if (state === 'loading') {
    return (
      <div style={{
        width: '100vw', height: '100vh', background: '#0a0a0e',
        display: 'flex', alignItems: 'center', justifyContent: 'center',
      }}>
        <span style={{ fontFamily: 'IBM Plex Mono, monospace', color: '#8a8a8a', fontSize: '13px', letterSpacing: '0.12em' }}>
          ACQUIRING SIGNAL...
        </span>
      </div>
    )
  }

  if (state === 'error') {
    return (
      <div style={{
        width: '100vw', height: '100vh', background: '#0a0a0e',
        display: 'flex', alignItems: 'center', justifyContent: 'center',
      }}>
        <span style={{ fontFamily: 'IBM Plex Mono, monospace', color: '#8a8a8a', fontSize: '13px', letterSpacing: '0.12em' }}>
          SIGNAL LOST
        </span>
      </div>
    )
  }

  return (
    <>
      <div
        ref={containerRef}
        style={{
          width: '100vw', height: '100vh',
          background: '#0a0a0e',
          position: 'absolute', top: 0, left: 0,
        }}
      />
      {state === 'empty' && (
        <div style={{
          position: 'absolute', top: '50%', left: '50%',
          transform: 'translate(-50%, -50%)',
          fontFamily: 'IBM Plex Mono, monospace',
          color: '#8a8a8a', fontSize: '13px', letterSpacing: '0.12em',
          pointerEvents: 'none',
        }}>
          NO EVENTS IN WINDOW
        </div>
      )}
    </>
  )
}
```

## ANTI-PATTERN GUARDS

1. **No raw Three.js** — always use `globe.gl` primitives (`arcsData`, `pointsData`)
2. **No pure white/black** — `#ffffff` and `#000000` never appear in any CSS
3. **No system fonts** — Bebas Neue, IBM Plex Mono, Instrument Serif only
4. **Animation floor** — nothing animates under 300ms
5. **Globe.tsx must only run on the client** — `dynamic(() => import('../components/Globe'), { ssr: false })` lives in page.tsx. Globe.tsx itself is a normal `'use client'` component. Do not add a second `dynamic()` wrapper inside Globe.tsx. Do not execute globe.gl code during SSR.
6. **Globe coloring via texture, not mesh mutation** — use `globeImageUrl` with a dark texture. Do NOT poke internal mesh properties via `globe?.()?.material`.
7. **Display state from filtered set** — derive `empty`/`ready` state AFTER applying time and type filters, not from raw fetch result.
8. **`visibleTypes` empty Set = zero events** — `size === 0` means all toggles off, render nothing. Not the same as `undefined` (which means no filter).
9. **Tailwind for layout only** — all color, typography, effects via CSS custom properties
10. **Globe canvas fills 100vw × 100vh** — `globals.css` resets all margin/padding; globe container div is `width: 100vw; height: 100vh`.
11. **Cleanup on unmount** — `clearInterval`, `removeEventListener` in `useEffect` return. No dangling timers.
